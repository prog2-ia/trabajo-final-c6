Clase base: Contenido:
    Atributos:
        titulo, fecha_lanzamiento, duracion (en segundos), genero (str | list[str]), artista (str o Artista), estado_reproduccion (bool), reproducciones (int).
    Métodos:
        reproducir(), pausar(), detener()
        formatear_duracion() -> "mm:ss"
        validar() (duración ≥ 0, título no vacío, etc.)
Hereda: Cancion, Album, ListaReproduccion.

Cancion(Contenido):
    Atributos adicionales:
        archivo_audio (ruta opcional), estado_animo (opcional), discografica o album (elige uno), bitrate/formato (opcional).
    Métodos:
        validar_archivo() (existe, extensión válida)
        mostrar_info() (extiende base)

Album(Contenido):
    Atributos:
        artista, tipo (álbum/EP/single/compilación), discografica, lista_canciones: list[Cancion].
    Métodos:
        agregar_cancion(c), eliminar_cancion(c)
        calcular_duracion() (suma de canciones, actualiza duracion)
        obtener_num_canciones()

ListaReproduccion(Contenido):
    Atributos:
        lista_canciones: list[Cancion | str], duracion_maxima (segundos, opcional), criterios (género predominante, estado de ánimo).
    Métodos:
        agregar_cancion(c), eliminar_cancion(c)
        eliminar_duplicados()
        mezclar() (shuffle determinista opcional con semilla)
        filtrar_por_genero(g), filtrar_por_estado(e)
        ajustar_a_duracion_maxima()
